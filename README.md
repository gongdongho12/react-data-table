# react-data-table

> Made with create-react-library

## Install

```bash
npm install --save @storelink-io/react-data-table
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from '@storelink-io/react-data-table'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
